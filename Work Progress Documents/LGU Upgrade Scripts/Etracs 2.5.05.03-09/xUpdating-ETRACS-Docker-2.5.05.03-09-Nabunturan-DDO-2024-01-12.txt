04. Stop all containers from the old docker 

    $  cd ~/docker/bin
    
    $  sh stop-gdx.sh
    
    $  sh stop.sh
    

    $  docker container prune -f
    
    $  docker volume prune -f

    $  docker image prune -f


05. Rename the old docker folder 

    $  cd
    
    $  mv docker  docker-old-2024-01-12


06. Rename the new docker folder as the official "docker"

    $  cd
    
    $  mv docker-new  docker


07. Rerun the updating of docker images

    $  cd ~/docker/bin

    $  sh update-docker-images.sh


08. Run the containers

    $  sh start.sh


09. Run GDX container

    $  sh restart-gdx.sh

          when you see colored text messages
